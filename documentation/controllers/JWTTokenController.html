import { BadRequestException, Body, Controller, Post, Req, Res, Get } from '@nestjs/common';
import { JWTTokenService } from './jwtToken.service';
import * as bcrypt from 'bcrypt';
import { JwtService } from '@nestjs/jwt';
import { Response, Request } from 'express';

@Controller('jwt-token')
export class JWTTokenController {
    constructor(
        private readonly jwtTokenService: JWTTokenService,
        private jwtService: JwtService
    ) { }

    @Post('/register')
    async register(
        @Body('name') name: string,
        @Body('email') email: string,
        @Body('password') password: string
    ) {

        const hashedPassword = await bcrypt.hash(password, 20)
        const user = await this.jwtTokenService.register({
            name,
            email,
            password: hashedPassword
        })
        delete user.password;
        return user;
    }

    @Post("login")
    async login(
        @Body('email') email: string,
        @Body('password') password: string,
        @Res({ passthrough: true }) response: Response
    ) {

        const user = await this.jwtTokenService.login({ email });

        if (!user) {
            throw new BadRequestException('Invalid Credentials');
        }
        if (!await bcrypt.compare(password, user.password)) {
            throw new BadRequestException('invalid Credentials');
        }


        const jwt = await this.jwtService.signAsync({ id: user.id });
        response.cookie('jwt', jwt, { httpOnly: true });
        console.log("token: ", jwt)
        // return {
        //     message: 'Success'
        // };

        return jwt;
    }
